import 'dart:io';
import 'dart:convert';

// Student Class
class Student {
  String name;
  int rollNo;
  Map<String, int> marks;

  Student(this.name, this.rollNo, this.marks);

  int get totalMarks => marks.values.reduce((a, b) => a + b);
  double get percentage => totalMarks / marks.length;

  String get result => percentage >= 40 ? "Pass" : "Fail";

  void display() {
    print("\n--- Student Details ---");
    print("Name      : $name");
    print("Roll No   : $rollNo");
    print("Marks     : $marks");
    print("Total     : $totalMarks");
    print("Percentage: ${percentage.toStringAsFixed(2)}%");
    print("Result    : $result");
  }
}

// Global Data Structures
List<Student> studentList = [];
Set<int> usedRollNumbers = {};

void main() {
  while (true) {
    print("\n===== Student Management Menu =====");
    print("1. Add Student");
    print("2. View All Students");
    print("3. Export to JSON");
    print("4. Exit");
    stdout.write("Enter choice: ");
    String? choice = stdin.readLineSync();

    switch (choice) {
      case '1':
        addStudent();
        break;
      case '2':
        viewStudents();
        break;
      case '3':
        exportToJson();
        break;
      case '4':
        print("Exiting program...");
        return;
      default:
        print("❌ Invalid choice. Try again.");
    }
  }
}

// ✅ Normal Function with try-catch
void addStudent() {
  try {
    stdout.write("Enter student name: ");
    String name = stdin.readLineSync()!.trim();

    stdout.write("Enter roll number: ");
    int roll = int.parse(stdin.readLineSync()!.trim());

    if (usedRollNumbers.contains(roll)) {
      print("❌ Roll number already exists!");
      return;
    }

    Map<String, int> marks = {};
    List<String> subjects = ["English", "Math", "Science"];
    for (var subject in subjects) {
      stdout.write("Enter marks of $subject: ");
      int mark = int.parse(stdin.readLineSync()!.trim());
      if (mark < 0 || mark > 100) {
        print("❌ Marks must be between 0 and 100");
        return;
      }
      marks[subject] = mark;
    }

    Student s = Student(name, roll, marks);
    studentList.add(s);
    usedRollNumbers.add(roll);
    print("✅ Student added successfully!");
  } catch (e) {
    print("⚠️ Invalid input. Please try again.");
  }
}

// ✅ Arrow function
void viewStudents() =>
    studentList.isEmpty
        ? print("No students yet.")
        : studentList.forEach((s) => s.display());

// ✅ Using Map + JSON
void exportToJson() {
  try {
    List<Map<String, dynamic>> data = studentList.map((s) => {
          "name": s.name,
          "rollNo": s.rollNo,
          "marks": s.marks,
          "total": s.totalMarks,
          "percentage": s.percentage,
          "result": s.result
        }).toList();

    File file = File("students.json");
    file.writeAsStringSync(jsonEncode(data), mode: FileMode.write);
    print("📁 Exported to students.json successfully!");
  } catch (e) {
    print("⚠️ Could not export file. Try again.");
  }
}
